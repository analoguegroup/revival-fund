                    <div class="faq-answer">
                        <p>We work directly with living researchers or the estates and intellectual heirs of deceased scholars.                     </div>
                </div>

                <div class="faq-item">
                    <div class="faq-question" onclick="toggleFaq(this)">
                        <span>What does "publishing-led" mean in practice?</span>
                        <div class="faq-icon"></div>
                    </div>
                    <div class="faq-answer">
                        <p>Unlike traditional academic recovery projects that begin with conferences or journal articles, we start with the artifact. We design the book, dossier, or archive first, treating the physical or digital object as the primary site of interpretation. The goal is to create something that feels inevitable—work that should have been in circulation all along.</p>
                    </div>
                </div>

                <div class="faq-item">
                    <div class="faq-question" onclick="toggleFaq(this)">

                    </div>
                    <div class="faq-answer">

If you know of neglected research that deserves revival, please send a brief description (500 words maximum) to proposals@researchrevival.org.                 </div>

    <script>
        function showPage(pageId) {
            // Hide all pages
            document.querySelectorAll('.page').forEach(page => {
                page.classList.remove('active');
            });
            
            // Show selected page
            document.getElementById(pageId).classList.add('active');
            
            // Update nav active state
            document.querySelectorAll('.nav-links a').forEach(link => {
                link.classList.remove('active');
            });
            if(pageId !== 'home') {
                document.getElementById('nav-' + pageId).classList.add('active');
            }
            
            // Scroll to top
            window.scrollTo(0, 0);
        }

        function toggleFaq(element) {
            const faqItem = element.parentElement;
            const isOpen = faqItem.classList.contains('open');
            
            // Close all other items
            document.querySelectorAll('.faq-item').forEach(item => {
                item.classList.remove('open');
            });
            
            // Toggle current item
            if (!isOpen) {
                faqItem.classList.add('open');
            }
        }

        // Smooth scroll behavior for internal links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
            });
        });
    </script>
